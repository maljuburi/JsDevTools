    // initialize map veriable
    
    var map;
     
    // init map
    
    function initMap() {
        // Create a map object and specify the DOM element for display. 
        map = new google.maps.Map(document.getElementById('map'), {
            center: {
                lat: 43,
                lng: -76
            },
            zoom: 7
        });
        
        
        var marker = new google.maps.Marker({
            position: map.center,
            map: map
        });
        
        infoWindow = new google.maps.InfoWindow;
    }
    
    function findMe(){
        
        // If browser support Geolocation
        
        if (navigator.geolocation) {
            navigator.geolocation.getCurrentPosition(function (position){
                var pos = {
                    lat: position.coords.latitude,
                    lng: position.coords.longitude
                };
                
                // Popup marker on my location
                
                infoWindow.setPosition(pos);
                infoWindow.setContent("
                <h1> ☉ You are here!</h1>
                <hr> <br>
                <h4>Latitude : "+pos.lat+"</h4>
                <br>
                <h4>Longitude : "+pos.lng+"</h4>"
                );

                infoWindow.open(map);
                map.setCenter(pos);
                map.setZoom(15);
                
                document.getElementById("result").innerHTML= "
                <h4>Latitude: " + pos.lat + "</h4>";

    // Onclick Find my location 
    
    document.getElementById("findMe").addEventListener('click', function () {
